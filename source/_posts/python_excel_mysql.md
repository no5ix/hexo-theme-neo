            # 发生错误时回滚

            print("Error: clear_label_by_material_id material_id=%d" % material_id)

    def get_label_name_2_id_map(self):
        # SQL 查询语句
        sql = "SELECT * FROM label"
        ret_map = {}
        try:
            # 执行SQL语句
            self.cur.execute(sql)
            # 获取所有记录列表
            results = self.cur.fetchall()
            for row in results:
                _id = row[0]
                _name = row[1]
                ret_map[_name] = _id

        except:
            # 发生错误时回滚
            # self.conn.rollback()
            print("Error: get_label_id_2_name_map")
        return ret_map
